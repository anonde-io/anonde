# HuggingFace Datasets — used by every corpus loader that streams gold

# at the data-fetch step. Not pinned: dataset access is auto-converted
# parquet refs which are stable across recent versions.

gliner==0.2.26

# transformers — used by bench/runners/openai_pf.py (the openai-pf
# cell, excluded from the CI subset but kept loadable for local
# debugging) and pulled transitively by gliner.
transformers

pyyaml
